# Sampling variance for log of the mean (Paul et al., 2008) to use in multivariate model
wb_yld = wb_yld %>% 
  mutate(log_yld = log(mean_yld))

wb_yld$log_var_yld <- with(wb_yld, vi_yld / (n * mean_yld^2))
